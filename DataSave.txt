class PersonData
{
	private String name;
	private int age;
	private String gen
	private int phoneNum;
	pirvate String add;

	public PersonData()
	{}
	
	public PersonData(String name, int age, String gen, int phoneNum, String add)
	{
		this.name=name;
		this.age=age;
		this.gen=gen;
		this.phoneNum=phoneNum;
		this.add=add;
	}
	@Override
	public String toString()
	{
		return name+"("+age+", "+gen+")"+" "+phoneNum+" "+add;
	}

	public void setName(String name)
	{
		this.name=name;
	}
	public getName()
	{
		return name;
	}

	public void setAge(int age)
	{
		this.age=age;
	}
	public getAge()
	{
		return age
	}
	public void setGen(String gen
	{
		this.gen=gen;
	}
	public getGen()
	{
		return gen;
	}
	public void setPhoneNum(int phoneNum)
	{
		this.phoneNum=phoneNum;
	}
	public getPhoneNum()
	{
		return phoneNum;
	}
	public void setAdd(String add)
	{
		this.add=add;
	}
	public getAdd()
	{
		return add;
	}
}
